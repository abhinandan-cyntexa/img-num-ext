'use strict';

// ─── Bootstrap ────────────────────────────────────────────────────────────────
// Reference: tableau-viz/extensions/bubble-chart/chart.js L9-13
tableau.extensions.initializeAsync().then(() => {
  const ws = tableau.extensions.worksheetContent.worksheet;
  console.log('[img-num-ext] worksheet:', ws.name);

  // Register SummaryDataChanged listener EXACTLY ONCE (never inside render).
  ws.addEventListener(tableau.TableauEventType.SummaryDataChanged, () => render(ws));

  // Initial render — SummaryDataChanged does not fire on load.
  render(ws);
}).catch(err => showStatus('Init error: ' + (err.message || String(err))));

// ─── Render pipeline ──────────────────────────────────────────────────────────
// Reference: tableau-viz/extensions/bubble-chart/chart.js L22-31
async function render(worksheet) {
  showStatus('Loading...');
  try {
    const [vizSpec, dataTable] = await Promise.all([
      worksheet.getVisualSpecificationAsync(),
      fetchData(worksheet),
    ]);

    const parsed = parseFields(dataTable, vizSpec);

    // Phase 1: verify the data pipeline via console logs. No card rendering yet.
    console.log('[img-num-ext] row count:', dataTable.data.length);
    console.log('[img-num-ext] columns:', dataTable.columns.map(c => c.fieldName));
    console.log('[img-num-ext] columns (caption):', dataTable.columns.map(c => c.fieldCaption));
    console.log('[img-num-ext] imageField fieldName:', parsed.imageField);
    console.log('[img-num-ext] numberField fieldName:', parsed.numberField);
    console.log('[img-num-ext] imgIdx:', parsed.imgIdx, 'numIdx:', parsed.numIdx);

    if (parsed.imgIdx === -1 || parsed.numIdx === -1) {
      showStatus('Loaded — please map Image URL and Number fields');
    } else {
      showStatus('Loaded — ' + dataTable.data.length + ' rows');
    }
  } catch (err) {
    showStatus('Error: ' + (err.message || String(err)));
  }

// ─── Data fetch with try/finally reader release ───────────────────────────────
// Reference: tableau-viz/extensions/bubble-chart/chart.js L15-19
// CRITICAL: reader.releaseAsync() in finally prevents cursor leak.
async function fetchData(worksheet) {
  const reader = await worksheet.getSummaryDataReaderAsync(undefined, { ignoreSelection: true });
  try {
    return await reader.getAllPagesAsync();
  } finally {
    await reader.releaseAsync();
  }
}

// ─── Encoding → column index resolution ───────────────────────────────────────
// Reference: tableau-viz/extensions/bubble-chart/chart.js L60-76
// CRITICAL: join key is fieldName (NOT fieldCaption) on BOTH sides.
function parseFields(dataTable, vizSpec) {
  const marksSpec = vizSpec.marksSpecificationCollection[0];
  if (!marksSpec) {
    // Graceful degradation: unmapped worksheet is valid starting state.
    return { imageField: null, numberField: null, imgIdx: -1, numIdx: -1 };
  }

  const encodings = marksSpec.encodingCollection;

  function resolveFieldName(encodingId) {
    const enc = encodings.find(e => e.id === encodingId);
    return enc?.fieldCollection?.[0]?.fieldName ?? null;
  }

  // Encoding IDs must match <encoding id="..."> in img-num-ext-local.trex exactly.
  const imageField  = resolveFieldName('image-url');
  const numberField = resolveFieldName('number-value');

  const colIndex = Object.fromEntries(
    dataTable.columns.map(c => [c.fieldName, c.index])
  );

  const imgIdx = imageField && imageField in colIndex ? colIndex[imageField] : -1;
  const numIdx = numberField && numberField in colIndex ? colIndex[numberField] : -1;

  return { imageField, numberField, imgIdx, numIdx };
}

// ─── UI helper ────────────────────────────────────────────────────────────────
function showStatus(msg) {
  const el = document.getElementById('card-container');
  if (el) el.textContent = msg;
}

- Pitfall 3 (wrong API path): use `vizSpec.marksSpecificationCollection[0].encodingCollection` — NOT `vizSpec.marksSpecificationsByEncoding`.
